#!/bin/bash -x
for file in `ls *.txt`
do
    foldername=`echo $file | awk -F. '{print $1}'`;
    echo $foldername
    if [ -d $foldername ]
     then
           rm -r $foldername
    fi
    mkdir $foldername
    cp $file $foldername
done

#!/bin/bash -x
for file in `ls *.log.1`
do
  name=`echo $file | awk -F. '{print $1}'`
  echo $name
  today=$(date +"%d%m%y")
  echo $today
  xyz=$name-$today".log"
  echo $xyz
  mv $file $xyz
done
